# Build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

COPY ["BusService/BusService.csproj", "BusService/"]
RUN dotnet restore "BusService/BusService.csproj"

COPY . .
WORKDIR "/src/BusService"
RUN dotnet publish "BusService.csproj" -c Release -o /app/publish

# Run stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app
COPY --from=build /app/publish .
EXPOSE 8080
ENV ASPNETCORE_URLS=http://+:8080
ENTRYPOINT ["dotnet", "BusService.dll"]
